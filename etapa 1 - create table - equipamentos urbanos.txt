-- Cria tabela de postos da URBS
CREATE TABLE tcc_curitiba_maria_postos_urbs (
    nome TEXT NOT NULL,
    endereco TEXT NOT NULL,
    lat DOUBLE PRECISION,
    lon DOUBLE PRECISION
);

ALTER TABLE tcc_curitiba_maria_postos_urbs
ADD COLUMN geom geometry(Point, 31982);

UPDATE tcc_curitiba_maria_postos_urbs
SET geom = ST_Transform(ST_SetSRID(ST_MakePoint(lon, lat), 4326),31982)
